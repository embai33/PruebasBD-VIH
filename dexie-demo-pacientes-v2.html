// 1. Definir la base de datos
const db = new Dexie('PacientesDB');
db.version(1).stores({
  pacientes: '++id, nhc, nombre, edad, diagnostico, tratamiento, estado'
});

// 2. Crear (INSERT)
await db.pacientes.add({ nhc, nombre, edad, ... });

// 3. Leer todos (SELECT *)
const todos = await db.pacientes.toArray();

// 4. Buscar (WHERE)
const activos = await db.pacientes
  .where('estado').equals('activo').toArray();
